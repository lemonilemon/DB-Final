## 8. MongoDB Aggregation Queries You Can Run

### Most Active Users
```javascript
db.user_behavior.aggregate([
  { $match: {
      timestamp: {
        $gte: ISODate("2025-11-07T00:00:00Z")
      }
  }},
  { $group: {
      _id: "$user_id",
      total_actions: { $sum: 1 },
      actions: { $push: "$action_type" }
  }},
  { $sort: { total_actions: -1 } },
  { $limit: 10 }
])
```

### Peak Usage Hours
```javascript
db.api_usage.aggregate([
  { $project: {
      hour: { $hour: "$timestamp" },
      endpoint: 1
  }},
  { $group: {
      _id: "$hour",
      request_count: { $sum: 1 }
  }},
  { $sort: { _id: 1 } }
])
```

### Recipe Conversion Funnel
```javascript
db.user_behavior.aggregate([
  { $match: {
      resource_type: "recipe"
  }},
  { $group: {
      _id: {
        recipe_id: "$resource_id",
        action_type: "$action_type"
      },
      count: { $sum: 1 }
  }},
  { $group: {
      _id: "$_id.recipe_id",
      views: {
        $sum: {
          $cond: [{ $eq: ["$_id.action_type", "view_recipe"] }, "$count", 0]
        }
      },
      cooks: {
        $sum: {
          $cond: [{ $eq: ["$_id.action_type", "cook_recipe"] }, "$count", 0]
        }
      }
  }},
  { $project: {
      recipe_id: "$_id",
      views: 1,
      cooks: 1,
      conversion_rate: {
        $multiply: [
          { $divide: ["$cooks", "$views"] },
          100
        ]
      }
  }}
])
```
